function checkAnswer(inputId, correctAnswer) { const userAnswer = document.getElementById(inputId).value.trim(); const feedback = document.getElementById(`feedback${inputId.slice(-1)}`); if (userAnswer.toUpperCase() === correctAnswer.toUpperCase()) { feedback.textContent = 'Correto! Bem feito!'; feedback.style.color = 'green'; } else { feedback.textContent = 'Incorreto. Tente novamente.'; feedback.style.color = 'red'; } }
